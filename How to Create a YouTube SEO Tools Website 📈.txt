<script>
function generateFullPackage() {
    const topic = document.getElementById('input-topic').value;
    const channel = document.getElementById('cfg-channel').value || "Manula Nirwan";
    if(!topic) return alert("Enter a topic!");

    // Titles
    document.getElementById('titles-output').innerText = `1. ${topic} - The Honest Review You Need! (2026)\n2. ${topic} Review: Why It’s The Ultimate Choice?\n3. Is ${topic} Worth The Hype? (Insane Details)`;

    // Description
    const desc = `🔥 Welcome to ${channel}!\n\nIn this video, we explore ${topic}.\n\n🚀 What You'll Learn:\n✔️ Deep dive into ${topic}\n✔️ Expert analysis & performance\n✔️ Hidden details you missed\n\n⏱️ Timestamps:\n00:00 Intro\n00:35 Detailed Review\n02:10 Key Features\n04:45 Final Thoughts\n\n🎯 Why Watch?\nThis video gives you complete insights about ${topic}.\n\n📌 Related Searches:\n${topic} review, ${topic} 2026, ${channel} car reviews\n\n💬 Comment Your Opinion Below!\n👍 Like, Share & Subscribe for more content.\n\n#${topic.replace(/\s+/g, '')} #CarReview #ManulaNirwan #LuxuryCars`;
    document.getElementById('desc-output').innerText = desc;

    // Tags
    const tags = [topic, channel, "manula nirwan", "mrmanula cars", "luxury cars", "review", "2026", "technology"];
    const container = document.getElementById('tags-output');
    container.innerHTML = tags.map(t => `<span class="badge-tag">${t}</span>`).join('');

    // AI Prompt
    document.getElementById('ai-prompt-output').innerText = `AI Thumbnail Prompt: A professional, high-contrast, extreme close-up of ${topic}, dramatic lighting, "INSANE" text overlay, blurred city background, 8k resolution, YouTube thumbnail style.`;

    document.getElementById('output-board').style.display = 'block';
}
</script>
