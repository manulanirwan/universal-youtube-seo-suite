    <script>
        let generatedTagsArray = [];

        function cleanStringForSEO(str) {
            return str.toLowerCase()
                .replace(/[^a-z0-9\s-]/g, '')
                .replace(/\s+/g, '-')
                .replace(/-+/g, '-');
        }

        function generateSeoPackage() {
            const topic = document.getElementById('input-topic').value.trim();
            if (!topic) {
                alert("Please input a topic/keyword parameter string.");
                return;
            }

            // Extract dynamic context settings variables with fallbacks
            const chName = document.getElementById('cfg-channel-name').value.trim() || "Your Channel Name";
            const chUrl = document.getElementById('cfg-channel-url').value.trim() || "https://www.youtube.com/@YourChannel";
            const langKeywords = document.getElementById('cfg-lang-tags').value.trim() || "tech, tutorial, tips";
            const seoSlug = cleanStringForSEO(topic);

            // 1. Dynamic Titles Mapping Pipeline
            document.getElementById('title-opt-1').value = `${topic} | Step-by-Step Guide 🚀`;
            document.getElementById('title-opt-2').value = `${topic}? Fix This Instantly! 💡`;
            document.getElementById('title-opt-3').value = `Don't Miss This: Ultimate ${topic} Tutorial`;

            // 2. Structural Description Blueprint Compilation 
            const descriptionBlockText = `⚡ Welcome to ${chName}!

In this video, we cover everything you need to know about "${topic}". This comprehensive tutorial guide breaks down each step clearly so you can implement it with ease.

🔥 Video Timeline Chapters:
00:00 - Introduction & Setup Overview
01:15 - Required Prerequisites Configuration
03:00 - Detailed Step-by-Step Implementation
06:45 - Essential Performance Adjustments & Common Errors Fix
09:15 - Final Summary & Important Takeaways

📌 Subscribe to the channel for more useful guides and updates:
👉 ${chUrl}

👍 If you found this content helpful, please leave a LIKE, drop a COMMENT, and SHARE the video!

🔎 Search Keywords Covered:
${topic.toLowerCase()}, ${langKeywords}, how to, educational walkthrough, technology setup guide`;

            document.getElementById('output-description').innerText = descriptionBlockText;

            // 3. Automated Video Tags Core Logic Matrix (Strictly cap below 500 characters)
            let baseKeywordsRaw = [
                "tutorial", "how to", "guide", "step by step", "tips", "tricks", "walkthrough", "course"
            ];
            
            // Clean custom parameter keywords inputs
            langKeywords.split(',').forEach(k => {
                if(k.trim()) baseKeywordsRaw.push(k.trim().toLowerCase());
            });

            // Splice keyword structures to form relevant query strings
            let baseTags = [];
            baseTags.push(topic.toLowerCase());
            baseTags.push(chName.toLowerCase());
            
            const words = topic.toLowerCase().split(' ');
            if(words.length > 2) {
                baseTags.push(`${words[0]} ${words[1]} ${words[2]}`);
                baseTags.push(`${words[0]} ${words[1]}`);
            }

            baseKeywordsRaw.forEach(kw => {
                if(words.length > 1) {
                    baseTags.push(`${words[0]} ${kw}`);
                    baseTags.push(`${topic.toLowerCase()} ${kw}`);
                }
            });

            // Compile tag badges and compute array string length boundaries
            generatedTagsArray = [];
            let characterAccumulator = 0;
            const cloudContainer = document.getElementById('output-tags-cloud');
            cloudContainer.innerHTML = "";

            for(let i = 0; i < baseTags.length; i++) {
                let currentTag = baseTags[i];
                // Check safety margin constraints for YouTube forms metadata parameters
                if ((characterAccumulator + currentTag.length + 2) <= 490) {
                    generatedTagsArray.push(currentTag);
                    characterAccumulator += currentTag.length + 2;

                    const tagSpan = document.createElement('span');
                    tagSpan.className = "badge-tag";
                    tagSpan.innerText = currentTag;
                    cloudContainer.appendChild(tagSpan);
                } else {
                    break;
                }
            }
            document.getElementById('tag-char-counter').innerText = `${characterAccumulator} / 500 Characters`;

            // 4. Client AI Prompt Matrix Ingestion Layout Generation
            const aiPromptText = `A professional high-click-through-rate YouTube thumbnail illustration for a modern technology tutorial video titled: "${topic}". 
The overall design utilizes a clean tech environment background with glowing servers, virtual network connectivity nodes, and sharp neon graphic accent layers. 
Features bold, contrasting, highly readable display typography centered on screen to ensure maximum clarity on mobile phone views. 
The aesthetic uses modern, vivid gradients with crisp vector asset layers, dramatic shadows, sharp focus detailing, and an ultra-high-resolution composition setup.`;

            document.getElementById('output-thumbnail-prompt').innerText = aiPromptText;
            document.getElementById('meta-filename-target').innerText = `${seoSlug}-thumbnail.jpg`;

            // Mount Visibility Actions across layouts
            document.getElementById('seo-output-board').classList.remove('hidden-element');
            document.getElementById('seo-output-board').scrollIntoView({ behavior: 'smooth' });
        }

        // Clipboards Engine Operations
        function copyValueFromId(elementId) {
            const el = document.getElementById(elementId);
            navigator.clipboard.writeText(el.value);
            alert("Title copied successfully!");
        }

        function copyDivContentContent(elementId) {
            const txt = document.getElementById(elementId).innerText;
            navigator.clipboard.writeText(txt);
            alert("Text block data copied to clipboard!");
        }

        function copyTagsToClipboard() {
            if(generatedTagsArray.length === 0) return;
            const tagsString = generatedTagsArray.join(', ');
            navigator.clipboard.writeText(tagsString);
            alert("Tags string array ready for deployment paste actions!");
        }
